\subsubsection{Exercise: Normalization of Tensor Product States}

Given two qubits in the states $\ket{v_{A}}$ and $\ket{v_{B}}$ respectively:
\begin{equation*}
    \ket{v_{A}} = a_{0}\ket{0} + a_{1}\ket{1}, \quad \ket{v_{B}} = b_{0}\ket{0} + b_{1}\ket{1}
\end{equation*}
And the combined state of the two qubits is given by the tensor product:
\begin{equation*}
    \ket{v_{A}v_{B}} = \ket{v_{A}} \otimes \ket{v_{B}} = a_{0}b_{0}\ket{00} + a_{0}b_{1}\ket{01} + a_{1}b_{0}\ket{10} + a_{1}b_{1}\ket{11}
\end{equation*}
Show that:
\begin{equation*}
    \left|a_0 b_0\right|^2 + \left|a_0 b_1\right|^2 + \left|a_1 b_0\right|^2 + \left|a_1 b_1\right|^2 = 1
\end{equation*}

\highspace
\begin{proof}
    To show that the sum of the squares of the coefficients equals 1, we start expanding the expression:
    \begin{align*}
        \left|a_0 b_0\right|^2 + \left|a_0 b_1\right|^2 + \left|a_1 b_0\right|^2 + \left|a_1 b_1\right|^2
    \end{align*}
    Using the property:
    \begin{equation*}
        \left|xy\right|^{2} = \left|x\right|^2 \left|y\right|^2
    \end{equation*}
    Applying this property to each term, we get:
    \begin{align*}
        &= \left|a_0\right|^2 \left|b_0\right|^2 + \left|a_0\right|^2 \left|b_1\right|^2 + \left|a_1\right|^2 \left|b_0\right|^2 + \left|a_1\right|^2 \left|b_1\right|^2 \\[.3em]
        &= \left|a_0\right|^2 \left(\left|b_0\right|^2 + \left|b_1\right|^2\right) + \left|a_1\right|^2 \left(\left|b_0\right|^2 + \left|b_1\right|^2\right)
    \end{align*}
    Since $\ket{v_{B}}$ is a valid quantum state and must be normalized, we have:
    \begin{equation*}
        \left|b_0\right|^2 + \left|b_1\right|^2 = 1
    \end{equation*}
    Substituting this back into our expression, we get:
    \begin{align*}
        &= \left|a_0\right|^2 \cdot 1 + \left|a_1\right|^2 \cdot 1 \\[.3em]
        &= \left|a_0\right|^2 + \left|a_1\right|^2
    \end{align*}
    Since $\ket{v_{A}}$ is also a valid quantum state and must be normalized, we have:
    \begin{equation*}
        \left|a_0\right|^2 + \left|a_1\right|^2 = 1
    \end{equation*}
    Therefore, we conclude that:
    \begin{equation*}
        \left|a_0 b_0\right|^2 + \left|a_0 b_1\right|^2 + \left|a_1 b_0\right|^2 + \left|a_1 b_1\right|^2 = 1
    \end{equation*}
    The combined state $\ket{v_{A}v_{B}}$ is also a valid quantum state and must be normalized, which is consistent with our result.
\end{proof}

\noindent
This exercise demonstrates that the \textbf{tensor product of two normalized quantum states is also normalized}, because the squared amplitudes factorize and the normalization conditions of the individual states ensure that the total probability remains 1.
