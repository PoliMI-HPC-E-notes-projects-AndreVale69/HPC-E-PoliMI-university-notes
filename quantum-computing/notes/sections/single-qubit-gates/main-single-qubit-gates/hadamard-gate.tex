\subsubsection{Hadamard Gate (H)}\label{subsubsec:hadamard-gate-h}

The \definitionWithSpecificIndex{Hadamard Gate (H)}{Single Qubit Gates: Hadamard Gate (S)}{} \textbf{changes the basis representation of the qubit}. More concretely:
\begin{equation*}
    H : \left\{ \ket{0}, \, \ket{1}\right\} \longleftrightarrow \left\{ \ket{+}, \, \ket{-}\right\}
\end{equation*}
It maps computational basis states into equal superpositions and vice versa, allowing phase information to be transformed into measurable probabilities.

\highspace
\begin{flushleft}
    \textcolor{Green3}{\faIcon{square-root-alt} \textbf{Matrix Representation}}
\end{flushleft}
\begin{equation}
    H = \dfrac{1}{\sqrt{2}} 
    \begin{pmatrix}
        1 & 1 \\ 1 & -1
    \end{pmatrix}
\end{equation}

\highspace
\begin{flushleft}
    \textcolor{Green3}{\faIcon{book} \textbf{Action on Basis States}}
\end{flushleft}
\begin{align}
    H \ket{0} &= \dfrac{1}{\sqrt{2}}\left(\ket{0} + \ket{1}\right) = \dfrac{\ket{0} + \ket{1}}{\sqrt{2}} = \ket{+} \\[.5em]
    H \ket{1} &= \dfrac{1}{\sqrt{2}}\left(\ket{0} - \ket{1}\right) = \dfrac{\ket{0} - \ket{1}}{\sqrt{2}} = \ket{-}
\end{align}
The $\ket{+}$ and $\ket{-}$ \textbf{states} are known as the \definition{Hadamard basis}, or \textbf{superposition states}, with \textbf{equal probability amplitudes} for $\ket{0}$ and $\ket{1}$.

\highspace
\begin{flushleft}
    \textcolor{Green3}{\faIcon{book} \textbf{Superposition and Measurement}}
\end{flushleft}
After applying $H$ to $\ket{0}$, the resulting qubit is:
\begin{equation*}
    \ket{\psi} = \dfrac{1}{\sqrt{2}}\left(\ket{0} + \ket{1}\right)
\end{equation*}
This state is a \textbf{superposition} of $\ket{0}$ and $\ket{1}$, meaning that the qubit does not have a definite classical value prior to measurement. Instead, it is described by \textbf{probability amplitudes} associated with each basis state.

\highspace
Upon measurement in the computational basis, the state collapses probabilistically:
\begin{itemize}
    \begin{multicols}{2}
        \item \textbf{Probability of 0}:
        \begin{equation*}
            \left|\dfrac{1}{\sqrt{2}}\right|^{2} = \dfrac{1}{2}
        \end{equation*}
        \item \textbf{Probability of 1}:
        \begin{equation*}
            \dfrac{1}{2}
        \end{equation*}
    \end{multicols}
\end{itemize}
After the measurement:
\begin{itemize}
    \item The qubit collapses to $\ket{0}$ with probability $\frac{1}{2}$, or to $\ket{1}$ with probability $\frac{1}{2}$.
    \item The original superposition is \textbf{destroyed}, and cannot be recovered.
\end{itemize}

On the Bloch sphere, the \textbf{Hadamard gate performs two sequential rotations}:
\begin{enumerate}
    \item \hl{Rotation around the $y$-axis by $\frac{\pi}{2}$ radians ($90\degree$): $R_y\!\left(\frac{\pi}{2}\right)$.}

    \noindent
    The $y$-axis stays fixed. So the states on the $y$-axis $\ket{i}$ and $\ket{-i}$ do not move. Instead, the $z$-axis rotated toward the $x$-axis: $\ket{0} \rightarrow \ket{+}$ and $\ket{1} \rightarrow -\ket{-}$. So the first step already moves the computational basis from the $z$-axis to the $x$ axis.

    We can verify the matrix representation of $R_y\!\left(\frac{\pi}{2}\right)$ by using the exponential of the Pauli $Y$ matrix (see \autopageref{eq:rotation-operator-general}):
    \begin{align*}
        R_y\left(\theta\right) &= e^{-i\frac{\theta}{2}Y} \\
        &= \cos\left(\frac{\theta}{2}\right)I - i\sin\left(\frac{\theta}{2}\right)Y \\
        &= \cos\left(\frac{\theta}{2}\right)\begin{pmatrix}
            1 & 0 \\ 0 & 1
        \end{pmatrix} -
        i \sin\left(\frac{\theta}{2}\right) \begin{pmatrix}
            0 & -i \\ i & 0
        \end{pmatrix} \\
        &= \begin{pmatrix}
            \cos\left(\frac{\theta}{2}\right) & -\sin\left(\frac{\theta}{2}\right) \\[.5em]
            \sin\left(\frac{\theta}{2}\right) & \cos\left(\frac{\theta}{2}\right)
        \end{pmatrix}
    \end{align*}
    Now, substituting $\theta = \frac{\pi}{2}$ gives us the expected matrix for the first step of the Hadamard decomposition:
    \begin{equation*}
        R_y\left(\frac{\pi}{2}\right) = \dfrac{1}{\sqrt{2}} \begin{pmatrix}
            1 & -1 \\ 1 & 1
        \end{pmatrix}
    \end{equation*}
    \begin{itemize}
        \item Apply it to $\ket{0} = \begin{pmatrix} 1 \\ 0 \end{pmatrix}$:
        \begin{align*}
            R_y\left(\frac{\pi}{2}\right)\ket{0} &= \dfrac{1}{\sqrt{2}} \begin{pmatrix}
                1 & -1 \\ 1 & 1
            \end{pmatrix} \begin{pmatrix}
                1 \\ 0
            \end{pmatrix} \\[.3em]
            &= \dfrac{1}{\sqrt{2}} \begin{pmatrix}
                1 \\ 1
            \end{pmatrix} \\[.3em]
            &= \dfrac{1}{\sqrt{2}} \left(\begin{pmatrix}
                1 \\ 0
            \end{pmatrix} + \begin{pmatrix}
                0 \\ 1
            \end{pmatrix}\right) \\[.3em]
            &= \dfrac{1}{\sqrt{2}} \left(\ket{0} + \ket{1}\right) \\[.3em]
            &= \dfrac{\ket{0} + \ket{1}}{\sqrt{2}} \\[.3em]
            &= \ket{+}
        \end{align*}
        So $\ket{0}$, the north pole of the Bloch sphere, is rotated to the $+x$-axis, which is the $\ket{+}$ state.
        \item Apply it to $\ket{1} = \begin{pmatrix} 0 \\ 1 \end{pmatrix}$:
        \begin{align*}
            R_y\left(\frac{\pi}{2}\right)\ket{1} &= \dfrac{1}{\sqrt{2}} \begin{pmatrix}
                1 & -1 \\ 1 & 1
            \end{pmatrix} \begin{pmatrix}
                0 \\ 1
            \end{pmatrix} \\[.3em]
            &= \dfrac{1}{\sqrt{2}} \begin{pmatrix}
                -1 \\ 1
            \end{pmatrix} \\[.3em]
            &= \dfrac{1}{\sqrt{2}} \left(\begin{pmatrix}
                0 \\ 1
            \end{pmatrix} - \begin{pmatrix}
                1 \\ 0
            \end{pmatrix}\right) \\[.3em]
            &= \dfrac{1}{\sqrt{2}} \left(\ket{1} - \ket{0}\right) \\[.3em]
            &= -\dfrac{\ket{0} - \ket{1}}{\sqrt{2}} \\[.3em]
            &= -\ket{-}
        \end{align*}
        This means that $\ket{1}$, the south pole of the Bloch sphere, moves to the $-x$-axis, but with a global phase of $-1$. Since global phases do not affect measurement outcomes, we can ignore it and say that $\ket{1}$ is effectively rotated to the $-x$-axis, which is the $\ket{-}$ state. In simple terms: $-\ket{-} \equiv \ket{-}$ up to a global phase.
    \end{itemize}

    \item \hl{Rotation around the $x$-axis by $\pi$ radians ($180\degree$).} This step is represented by the operator $R_x(\pi)$, which can be expressed as:
    \begin{equation*}
        R_x(\pi) = e^{-i\frac{\pi}{2}X} = \cos\left(\frac{\pi}{2}\right)I - i\sin\left(\frac{\pi}{2}\right)X = -iX
    \end{equation*}
    Applying $R_x(\pi)$ to the states on the $x$-axis ($\ket{+}$ and $\ket{-}$) does not change their position on the Bloch sphere, but it does introduce a phase factor (which becomes a global phase only when applied to the entire state):
    \begin{align*}
        R_x(\pi)\ket{+} &= -iX\ket{+} = -i\ket{+} \\
        R_x(\pi)\ket{-} &= -iX\ket{-} = -i(-\ket{-}) = i\ket{-}
    \end{align*}
    Applying $R_x(\pi)$ to the superposition state $a\ket{+} - b\ket{-}$ (obtained from the first rotation) gives us:
    \begin{align*}
        &\phantom{=} a\ket{+} - b\ket{-} \\[.3em]
        R_x(\pi)(a\ket{+} - b\ket{-}) &= a(-i\ket{+}) - b(i\ket{-}) \\[.3em]
        &= -ia\ket{+} - ib\ket{-} \\[.3em]
        &= -i(a\ket{+} + b\ket{-})
    \end{align*}
    Since global phases $-i$ do not affect measurement outcomes, we can ignore them and say that $\ket{+}$ and $\ket{-}$ remain on the $x$-axis. The second rotation does not change the position of the states on the Bloch sphere, but it is necessary to ensure that the transformation matches the Hadamard gate at the operator level, preserving the correct relative phase for all superpositions.

    \textcolor{Green3}{\faIcon{question-circle} \textbf{Why is the second rotation necessary?}} We \textbf{cannot} simply change $-\ket{-}$ to $\ket{-}$ inside a superposition, because that would change only one component and therefore change the \textbf{relative phase}. The fix comes from applying $R_x(\pi)$. After the first rotation, we have:
    \begin{equation*}
        a\ket{+} - b\ket{-}
    \end{equation*}
    Now, applying the Pauli-X operator using the rotation $R_x(\pi)$ (\autopageref{eq:rotation-operator-general}):
    \begin{equation*}
        R_x(\pi) = -iX = -i \begin{pmatrix}
            0 & 1 \\ 1 & 0
        \end{pmatrix} \quad \text{with} \quad X\ket{+} = \ket{+}, \quad X\ket{-} = -\ket{-}
    \end{equation*}
    Applying $R_x(\pi)$ to the superposition gives us:
    \begin{align*}
        R_x(\pi)(a\ket{+} - b\ket{-}) &= -iX(a\ket{+} - b\ket{-}) \\
        &= -i\left(aX\ket{+} - bX\ket{-}\right) \\
        &= -i\left(a\ket{+} - b(-\ket{-})\right) \\
        &= -i\left(a\ket{+} + b\ket{-}\right)
    \end{align*}
    Now the factor $-i$ multiplies the \textbf{entire state}, so it is a global phase:
    \begin{equation*}
        -i(a\ket{+} + b\ket{-}) \equiv a\ket{+} + b\ket{-}
    \end{equation*}
    So the important point is:
    \begin{equation*}
        a\ket{+} - b\ket{-} \not\equiv a\ket{+} + b\ket{-}
    \end{equation*}
    But:
    \begin{equation*}
        R_x(\pi)(a\ket{+} - b\ket{-}) = -i(a\ket{+} + b\ket{-}) \equiv a\ket{+} + b\ket{-}
    \end{equation*}
    % The $x$-rotation flips the sign of the $\ket{-}$ component because $\ket{-}$ is the $-1$ eigenstate of $X$.

    \textcolor{Green3}{\faIcon[regular]{lightbulb} \textbf{Insight.}} The second rotation \textbf{does not affect the geometric position} of the states, but \textbf{corrects the relative phase} structure, ensuring that the transformation is a valid unitary operator acting consistently on the entire Hilbert space.

\highspace
\begin{flushleft}
    \textcolor{Green3}{\faIcon{book} \textbf{Eigenvalues and Eigenstates}}
\end{flushleft}
The Hadamard operator has eigenvalues:
\begin{equation*}
    \lambda = +1, \hspace{2em} \lambda = -1
\end{equation*}
with corresponding eigenstates given by the solutions of:
\begin{equation*}
    H\ket{v} = \lambda \ket{v}
\end{equation*}
These eigenstates are superpositions of $\ket{0}$ and $\ket{1}$, and are aligned along an axis halfway between the $x$ and $z$ directions on the Bloch sphere. Precisely, the eigenvectors are:
\begin{align*}
    \ket{v_{+}} &= \cos\left(\dfrac{\pi}{8}\right)\ket{0} + \sin\left(\dfrac{\pi}{8}\right)\ket{1} \\
    \ket{v_{-}} &= \sin\left(\dfrac{\pi}{8}\right)\ket{0} - \cos\left(\dfrac{\pi}{8}\right)\ket{1}
\end{align*}
Where $\ket{v_{+}}$ corresponds to the eigenvalue $+1$ and $\ket{v_{-}}$ corresponds to the eigenvalue $-1$; the cosine and sine factors ensure normalization of the eigenstates.

\highspace
Unlike Pauli gates, the Hadamard eigenstates lie along a \textbf{diagonal axis between the computational basis $z$ and the Hadamard basis $x$}, reflecting its role as a \textbf{basis transformation} rather than a simple bit or phase flip.

\highspace
\begin{flushleft}
    \textcolor{Green3}{\faIcon{list-ul} \textbf{Key Properties}}
\end{flushleft}
\begin{enumerate}
    \item \textbf{Unitary}: $H^\dagger H = I$, so normalization is preserved.
    
    \item \textbf{Hermitian}: $H^\dagger = H$, therefore it corresponds to an observable.
    
    \item \textbf{Self-inverse}: $H^2 = I$. Applying the Hadamard gate twice returns the original state.
    
    \item \textbf{Basis transformation}: H maps the computational basis $\left\{\ket{0}, \ket{1}\right\}$ to the Hadamard basis $\left\{\ket{+}, \ket{-}\right\}$:
    \begin{equation*}
        H\ket{0} = \ket{+}, \hspace{2em} H\ket{1} = \ket{-}
    \end{equation*}
    
    \item \textbf{Interference enabler}: $H$ converts phase differences into amplitude differences, making them observable through measurement.
    
    \item \textbf{Axis mixing}: $H$ transforms Pauli operators as:
    \begin{equation*}
        HZH = X, \hspace{2em} HXH = Z
    \end{equation*}
\end{enumerate}
