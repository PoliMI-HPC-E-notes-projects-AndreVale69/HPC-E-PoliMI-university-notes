\subsection{Measurement}

In quantum computing, \textbf{measurement} is a special type of operation that differs fundamentally from unitary gates. While gates perform reversible transformations, \textbf{measurement is irreversible}. It allows us to \textbf{gain information about the state of a qubit}, but at the price of \textbf{collapsing the quantum state}.

\highspace
\begin{flushleft}
  \textcolor{Green3}{\faIcon{square-root-alt} \textbf{Measurement Operator: Matrix Form}}
\end{flushleft}
A \textbf{measurement operator} is a \textbf{non-unitary}, \textbf{non-invertible} matrix. For a measurement \textbf{along a specific direction}, represented by a ket \ket{k}, the associated \textbf{projector} is (outer product):
\begin{equation}\label{eq:measurement-operator}
  M_{k} = \proj{k}
\end{equation}
This is called a \definition{Projection Operator}, it projects any vector \textbf{onto the direction of} \ket{k}.

\highspace
\begin{flushleft}
  \textcolor{Red2}{\faIcon{exclamation-triangle} \textbf{Effect of Measurement}}
\end{flushleft}
Given a qubit \ket{\psi}, applying $M_{k}$ yields the \textbf{(unnormalized) projected vector}:
\begin{equation}\label{eq:measurement-projection}
  \ket{\psi_{k}} = M_{k} \ket{\psi} = \ket{k} \braket{k}{\psi} = \left(\ket{k} \bra{k}\right) \ket{\psi}
\end{equation}
To obtain the \textbf{new normalized state}, we \textbf{divide by the square root of the probability}:
\begin{equation}\label{eq:measurement-normalization}
  \ket{\psi_{k}^{\text{norm}}} =
  \dfrac{M_{k} \cdot \ket{\psi}}{\sqrt{\left\langle \psi \left| M_{k} \right| \psi \right\rangle}} =
  \dfrac{\braket{k}{\psi} \cdot \ket{k}}{\sqrt{\left|\braket{k}{\psi}\right|^{2}}} =
  \ket{k}
\end{equation}
So after measurement, the qubit \textbf{collapses} to \ket{k}, the \textbf{eigenstate corresponding to the measurement outcome}.

\highspace
\begin{flushleft}
  \textcolor{Green3}{\faIcon{square-root-alt} \textbf{Why Normalization is Needed}}
\end{flushleft}
After applying the measurement operator $M_k = \ket{k}\bra{k}$ (\autoref{eq:measurement-operator}) to a state $\ket{\psi}$ (\autoref{eq:measurement-projection}), we obtain:
\begin{equation*}
  \ket{\psi_k} = M_k \ket{\psi} = \ket{k}\braket{k}{\psi}
\end{equation*}
This vector is generally \textbf{not normalized}. Its norm is:
\begin{equation*}
  \|\ket{\psi_k}\|^2 = \bra{\psi} M_k \ket{\psi} = |\braket{k}{\psi}|^2 = p_k
\end{equation*}
Where $p_k$ is the \textbf{probability of obtaining outcome $k$}.

\begin{deepeningbox}[: Why $\|\ket{\psi_k}\|^2 = p_k$?]
  We want to show:
  \begin{equation*}
    \left\| \ket{\psi_{k}} \right\|^{2} = \bra{\psi} M_{k} \ket{\psi} = \left|\braket{k}{\psi}\right|^{2} = p_{k}
  \end{equation*}
  \begin{enumerate}
    \item \textbf{Definition of squared norm}. The squared norm of a ket is:
    \begin{equation*}
      \left\| \ket{\psi_{k}} \right\|^{2} = \braket{\psi_{k}}{\psi_{k}}
    \end{equation*}
    Since $\ket{\psi_k} = M_k \ket{\psi}$, we can write:
    \begin{equation*}
      \left\| \ket{\psi_{k}} \right\|^{2} = \underbrace{\left(M_k \ket{\psi}\right)^{\dagger}}_{\bra{\psi_k}} \underbrace{\left(M_k \ket{\psi}\right)}_{\ket{\psi_k}}
    \end{equation*}
    We have only expanded the definition of the squared norm and substituted $\ket{\psi_k}$.

    \item \textbf{Take the dagger $\dagger$}. The dagger of a product of operators is the product of the daggers in reverse order:
    \begin{equation*}
      \left(A B\right)^{\dagger} = B^{\dagger} A^{\dagger}
    \end{equation*}
    Applying this to $\left(M_k \ket{\psi}\right)^{\dagger}$:
    \begin{equation*}
      \left(M_k \ket{\psi}\right)^{\dagger} = \bra{\psi} M_k^{\dagger}
    \end{equation*}
    And substituting back into the expression for the squared norm:
    \begin{equation*}
      \left\| \ket{\psi_{k}} \right\|^{2} = \bra{\psi} M_k^{\dagger} M_k \ket{\psi}
    \end{equation*}
    
    \item \textbf{Use the Hermitian property of $M_k$} (explained later \autopageref{prop:hermitian}). Since $M_k$ is Hermitian, we have $M_k^{\dagger} = M_k$. Therefore:
    \begin{equation*}
      \left\| \ket{\psi_{k}} \right\|^{2} = \bra{\psi} M_k M_k \ket{\psi}
    \end{equation*}

    \item \textbf{Use the idempotent property of $M_k$} (explained later \autopageref{prop:idempotent}). Since $M_k$ is idempotent, we have $M_k^2 = M_k$. Therefore:
    \begin{equation*}
      \left\| \ket{\psi_{k}} \right\|^{2} = \bra{\psi} M_k \ket{\psi}
    \end{equation*}

    \item \textbf{Substitute the definition of $M_k$}. Since $M_k = \ket{k}\bra{k}$, we can write:
    \begin{equation*}
      \left\| \ket{\psi_{k}} \right\|^{2} = \bra{\psi} \left(\ket{k}\bra{k}\right) \ket{\psi} = \braket{\psi}{k} \braket{k}{\psi}
    \end{equation*}
    \item \textbf{Use the property of inner products}. The inner product $\braket{\psi}{k}$ is the complex conjugate of $\braket{k}{\psi}$:
    \begin{equation*}
      \braket{\psi}{k} = \overline{\braket{k}{\psi}}
    \end{equation*}
    Therefore (\autopageref{sec:magnitude}):
    \begin{equation*}
      \left\| \ket{\psi_{k}} \right\|^{2} = \overline{\braket{k}{\psi}} \braket{k}{\psi} = \left|\braket{k}{\psi}\right|^{2} = p_k
    \end{equation*}
    This is exactly the probability $p_k$ of obtaining outcome $k$ when measuring the state $\ket{\psi}$ with the measurement operator $M_k$.
  \end{enumerate}
\end{deepeningbox}

\highspace
\textcolor{Red2}{\faIcon{exclamation-triangle} \textbf{Important.}} The result of a measurement must be a valid quantum state, and therefore it must be \textbf{normalized} (i.e., have unit norm).

\highspace
To restore normalization, we divide the projected vector by its norm:
\begin{equation*}
  \ket{\psi_k^{\text{norm}}}
  =
  \frac{\ket{\psi_k}}{\|\ket{\psi_k}\|}
  =
  \frac{\ket{k}\braket{k}{\psi}}{\sqrt{|\braket{k}{\psi}|^2}}
\end{equation*}
Observe that:
\begin{equation*}
  \frac{\braket{k}{\psi}}{|\braket{k}{\psi}|}
\end{equation*}
Is a complex number of magnitude $1$, i.e., a \textbf{global phase}. \hl{Since global phases do not affect physical states}, we conclude:
\begin{equation*}
  \ket{\psi_k^{\text{norm}}} = \ket{k}
\end{equation*}

\highspace
\textcolor{Green3}{\faIcon[regular]{lightbulb} \textbf{Interpretation}} Measurement can be understood as a two-step process:
\begin{enumerate}
  \item \textbf{Projection}: extract the component of $\ket{\psi}$ along $\ket{k}$ (this determines the probability $p_k$).
  \item \textbf{Renormalization}: rescale the result to obtain a valid quantum state.
\end{enumerate}

\highspace
\begin{flushleft}
  \textcolor{Green3}{\faIcon{percentage} \textbf{Probability of Outcome}}
\end{flushleft}
The \textbf{probability of measuring the qubit in the state} \ket{k} is:
\begin{equation}\label{eq: Born rule}
  p_{k} = \left\langle \psi \left| M_{k} \right| \psi \right\rangle = \left|\braket{k}{\psi}\right|^{2} =
  \left| \overline{k}_{1} \cdot \psi_{1} + \overline{k_{2}} \cdot \psi_{2} \right|^{2}
\end{equation}
This is the \definition{Born rule}, a fundamental postulate of quantum mechanics. In quantum computing we have:
\begin{equation*}
  \begin{array}{rcl}
    p_{0} &=& \left\langle \psi \left|M_{0}\right| \psi \right\rangle = \left|\braket{0}{\psi}\right|^{2} = \left|1 \cdot \psi_{1} + \cancel{0 \cdot \psi_{2}} \right| \\ [.8em]
    %
    p_{1} &=& \left\langle \psi \left|M_{1}\right| \psi \right\rangle = \left|\braket{1}{\psi}\right|^{2} = \left|\cancel{0 \cdot \psi_{1}} + 1 \cdot \psi_{2} \right|
  \end{array}
\end{equation*}

\begin{remarkbox}[: $\left\langle \psi \left| M_{k} \right| \psi \right\rangle$]
  This is called a quadratic form in linear algebra, and in quantum mechanics it represents the \textbf{probability of the measurement outcome} $k$ when \textbf{measuring the qubit} \ket{\psi} with \textbf{measurement operator} $M_{k}$.

  Suppose a qubit $\psi$:
  \begin{equation*}
    \ket{\psi} = \begin{pmatrix}
      a \\
      b
    \end{pmatrix}
    \hspace{2em}
    \text{with } \left|a\right|^{2} + \left|b\right|^{2} = 1 \text{ (normalization condition)}
  \end{equation*}
  If we apply the measurement operator $M_{0}$:
  \begin{equation*}
    M_{0} \ket{\psi} =
    \begin{pmatrix}
      1 & 0 \\ 0 & 0
    \end{pmatrix}
    \begin{pmatrix}
      a \\ b
    \end{pmatrix}
    =
    \begin{pmatrix}
      a \\ 0
    \end{pmatrix}
  \end{equation*}
  And we finally calculate the bra \bra{\psi}:
  \begin{equation*}
    \bra{\psi} \begin{pmatrix}
      a \\ 0
    \end{pmatrix}
    =
    \begin{pmatrix}
      \overline{a} & \overline{b}
    \end{pmatrix}
    \begin{pmatrix}
      a \\ 0
    \end{pmatrix}
    = \overline{a} \cdot a = \left|a\right|^{2}
  \end{equation*}
  In other words, this is the probability of measuring 0: $p_{0}$.
\end{remarkbox}

\highspace
\begin{flushleft}
  \textcolor{Green3}{\faIcon{book} \textbf{Standard Basis Measurement}}
\end{flushleft}
In quantum computing, we usually measure in the \textbf{computational basis}\break $\left\{\ket{0}, \ket{1}\right\}$. The \textbf{projectors} are:
\begin{equation}
  \begin{array}{rclcl}
    M_{0} &=& \ket{0}\bra{0} &=& \begin{pmatrix}
      1 & 0 \\
      0 & 0
    \end{pmatrix} \\ [1em]
    M_{1} &=& \ket{1}\bra{1} &=& \begin{pmatrix}
      0 & 0 \\
      0 & 1
    \end{pmatrix}
  \end{array}
\end{equation}
For a \textbf{general qubit} $\ket{\psi} = a\ket{0} + b\ket{1}$, the \textbf{probabilities} are:
\begin{equation}
  p_{0} = \left|a\right|^{2} \hspace{2em} p_{1} = \left|b\right|^{2}
\end{equation}
After measurement, the qubit \textbf{collapses to \ket{0} with probability $\left|a\right|^{2}$}, or to \textbf{\ket{1} with probability $\left|b\right|^{2}$}.

\highspace
\begin{flushleft}
  \textcolor{Green3}{\faIcon{star} \textbf{Key Properties of Measurement Operator}}
\end{flushleft}
\begin{itemize}
  \item \phantomsection\label{prop:hermitian} \important{Hermitian}: $M_{k}^{\dagger} = M_{k}$, which means that $M_k$ \textbf{is equal to its conjugate transpose}. This implies that $M_k$ has real eigenvalues and orthogonal eigenvectors.
  \begin{align*}
    M_{k}^{\dagger} &= \left(\ket{k}\bra{k}\right)^{\dagger} \\
    &\downarrow \text{ use the rule } (A B)^{\dagger} = B^{\dagger} A^{\dagger} \\
    &= (\underbrace{\ket{k}}_{A}\underbrace{\bra{k}}_{B})^{\dagger} \\
    &= \left(\bra{k}\right)^{\dagger} \left(\ket{k}\right)^{\dagger} \\
    &= \ket{k}\bra{k} \\
    &= M_{k}
  \end{align*}
  \item \phantomsection\label{prop:idempotent} \important{Idempotent}: once applied, \textbf{applying it again does nothing}:
  \begin{equation*}
    M_{k}^{2} = M_{k}
  \end{equation*}
  A simple proof:
  \begin{equation*}
    M_{k}^{2} = \ket{k} \proj{k} \bra{k} = \ket{k}\bra{k} = M_{k}
  \end{equation*}
  This reflects that \textbf{after projection, the qubit is already in the state} \ket{k}.

  \item \important{Non-unitary} and \important{Non-reversible}: \hl{measurement \textbf{destroys superposition} and \textbf{cannot be undone}.}
\end{itemize}
